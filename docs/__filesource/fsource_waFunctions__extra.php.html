  23.  * Modo de uso:
  24.  * <code>
  25.  * <?php
  26.  * echo geraPaginacao('clientes.php', 6, 4, (checkGET('np')?$_GET['np']:1) );

  39.  * @param string $pURL

  49. function geraPaginacao( $pURL, $pTotal, $pLimite, $pCurrentPage = 1 )
  50. {
  51.     //Máximo de Páginas
  52.     $iMax = ceil( $pTotal / $pLimite );
  53.  
  54.     //Verifica se a url já possui o char ?, para gerar entre (? e &) antes do 'np'
  55.     $pURL .= ( ( strpos( $pURL, '?' ) === false ) ? "?" : "&amp;" );
  56.  
  57.     $tmpStr = "\n<div id=\"paginacao\">\n\t<ul>";
  58.  
  59.     for ( $iCounter = 1; $iCounter <= $iMax; $iCounter++ ) {
  60.         $tmpMark = ( ( $iCounter == $pCurrentPage ) ? "class=\"marcado\"" : "" );
  61.         $tmpStr .= "\n\t\t<li><a href=\"{$pURL}np={$iCounter}\" {$tmpMark}>{$iCounter}</a></li>";
  62.     }
  63.  
  64.     $tmpStr .= "\n\t</ul>\n</div>";
  65.  
  66.     return $tmpStr;
  67. }
